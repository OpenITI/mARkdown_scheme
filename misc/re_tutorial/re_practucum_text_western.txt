	02	03	12	13

	501-1234	234 1252
	652.2648	713-342-7452
	PE6-5000	653-6464x256

	runs	run

	12345	122345	111111
	113456	098097	109493
	510349	673452	005645

	ar?t, a[fr]?t, ar*t, ar+t, a.*t

	3: “arrrrt”	4: “aft”
